# py -3.13 code.py

import pygame
import random

pygame.init()

screen = pygame.display.set_mode((400, 400))

# Create 50 snowflakes at random positions
snow = [[random.randrange(400), random.randrange(400)] for _ in range(50)]

clock = pygame.time.Clock()  # animation control

while True:
    for e in pygame.event.get():
        if e.type == pygame.QUIT:
            pygame.quit()
            exit()

    # Fill background with black color
    screen.fill((0, 0, 0))

    for s in snow:
        pygame.draw.circle(screen, (255, 255, 255), s, 4)
        # (surface, color, center, radius)

        s[1] += 1  # Move snowflake downward

        # If snowflake goes below screen, reset it above
        if s[1] > 400:
            s[1] = random.randrange(-20, 0)
            s[0] = random.randrange(400)

    pygame.display.flip()  # Update display

    clock.tick(20)  # Control animation speed
